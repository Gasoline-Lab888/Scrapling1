## 7. Suggested hvac-ai-crawler YAML

```yaml
brand: imi_climate_control_imi_hydronic_enginee
brand_name: "IMI Climate Control / IMI Hydronic Engineering"
website: https://climatecontrol.imiplc.com/
entry_url: https://climatecontrol.imiplc.com/product-finder
seed_urls:
  - https://climatecontrol.imiplc.com/product-finder
max_pages: 300

url_scope:
  path_prefix: "/product-finder"

page_role_classification:
  category:
    indicators:
      - "products"
    recurse: true
  product:
    indicators:
      - "datasheet"
    recurse: false
  navigation:
    indicators:
      - "download"
    recurse: true

fields:
  product_name: "h1"
  downloads: "a[href$='.pdf']"

filters:
  download_link_selector: "a[href$='.pdf']"
  reject_url_patterns:
    - "/news/"
    - "/blog/"
    - "/events/"
    - "/careers/"
    - "/privacy/"
    - "/terms/"
    - "/cookie/"
    - "/contact/"
    - "/about/"
  global_download_threshold: 10
```
