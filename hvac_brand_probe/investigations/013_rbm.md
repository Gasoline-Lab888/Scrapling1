## 7. Suggested hvac-ai-crawler YAML

```yaml
brand: rbm
brand_name: "RBM"
website: https://rbm.eu/
entry_url: https://rbm.eu/en/products
seed_urls:
  - https://rbm.eu/en/products
max_pages: 300

url_scope:
  path_prefix: "/en/products/"

page_role_classification:
  category:
    indicators:
      - "products"
    recurse: true
  product:
    indicators:
      - "datasheet"
    recurse: false
  navigation:
    indicators:
      - "download"
    recurse: true

fields:
  product_name: "h1"
  downloads: "a[href$='.pdf']"

filters:
  download_link_selector: "a[href$='.pdf']"
  reject_url_patterns:
    - "/news/"
    - "/blog/"
    - "/events/"
    - "/careers/"
    - "/privacy/"
    - "/terms/"
    - "/cookie/"
    - "/contact/"
    - "/about/"
  global_download_threshold: 10
```
