## 7. Suggested hvac-ai-crawler YAML

```yaml
brand: far_rubinetterie
brand_name: "FAR Rubinetterie"
website: https://www.far.eu/
entry_url: https://www.far.eu/index_en.php
seed_urls:
  - https://www.far.eu/index_en.php
max_pages: 300

url_scope:
  path_prefix: "/index_en.php"

page_role_classification:
  category:
    indicators:
      - "products"
    recurse: true
  product:
    indicators:
      - "datasheet"
    recurse: false
  navigation:
    indicators:
      - "download"
    recurse: true

fields:
  product_name: "h1"
  downloads: "a[href$='.pdf']"

filters:
  download_link_selector: "a[href$='.pdf']"
  reject_url_patterns:
    - "/news/"
    - "/blog/"
    - "/events/"
    - "/careers/"
    - "/privacy/"
    - "/terms/"
    - "/cookie/"
    - "/contact/"
    - "/about/"
  global_download_threshold: 10
```
